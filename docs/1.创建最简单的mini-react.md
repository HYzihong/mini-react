# 1. 创建最简单的 mini-react

### v1 传统 js 控制 dom 节点

我们需要创建一个文本节点，并且在页面显示

我们需要一个父节点，完后把文本节点的插入进去

```javascript
// 在页面创建一个文本显示出来

// 1. 创建一个父节点，插入根节点下
const dom = document.createElement("div");
dom.id = "root";
dom.style = "color:pink;";
document.querySelector("#root").append(dom);

// 2. 创建一个文本节点
const textNode = document.createTextNode("");
textNode.nodeValue = "app";

// 3. 把文本节点插入父节点下
dom.append(textNode);
```

### 动态创建 dom

#### 第一步,把创建 vdom 过程的抽离成函数

```javascript
// 在页面创建一个文本显示出来
// 使用vdom =》 js Object => 动态创建

function createTextNode(text) {
  return {
    type: "TEXT_ELEMENT",
    props: {
      nodeValue: text,
      children: [],
    },
  };
}

function createElement(type, props, ...children) {
  return {
    type,
    props: {
      ...props,
      children,
    },
  };
}

const textEl = createTextNode("app");
const App = createElement("app", { id: "app", style: "color:pink;" }, textEl);
```

#### 把创建 dom 的过程抽离成函数

找规律

```javascript
// 1. 创建一个父节点，插入根节点下
const dom = document.createElement(el.type); // 创建节点
dom.id = el.props.id; // 添加 props
dom.style = el.props.style; // 添加 props
document.querySelector("#root").append(dom); // 导入父节点

// 2. 创建一个文本节点
const textNode = document.createTextNode(""); // 创建节点
textNode.nodeValue = textEl.props.nodeValue; // 添加 props
// 3. 把文本节点插入父节点下
dom.append(textNode); // 导入父节点
```

1.  创建一个父节点，插入根节点下
2.  添加 props
3.  导入到 父节点

```javascript
// 在页面创建一个文本显示出来
// 使用vdom =》 js Object => 动态创建

function createTextNode(text) {
  return {
    type: "TEXT_ELEMENT",
    props: {
      nodeValue: text,
      children: [],
    },
  };
}

function createElement(type, props, ...children) {
  return {
    type,
    props: {
      ...props,
      children: children.map((child) => {
        // 处理纯文本节点
        return typeof child === "string" ? createTextNode(child) : child;
      }),
    },
  };
}

function render(el, container) {
  console.log("render el =>", el);
  console.log("render container =>", container);
  // 1. 判断类型来创建dom
  const dom =
    el.type != "TEXT_ELEMENT"
      ? document.createElement(el.type)
      : document.createTextNode("");

  //  2. 添加props
  Object.keys(el.props).forEach((key) => {
    if (key != "children") {
      dom[key] = el.props[key];
    }
  });
  // 处理children
  const childrens = el.props.children;
  childrens.forEach((child) => {
    render(child, dom);
  });

  // 3. 插入父节点中
  container.append(dom);
}

// 未处理单独处理文本节点
const textEl = createTextNode("app");
const App = createElement("app", { id: "app", style: "color:pink;" }, textEl);

// 特殊处理文本节点
const App = createElement("app", { id: "app", style: "color:pink;" }, "app");

// 传入多个子节点
const App = createElement(
  "app",
  { id: "app", style: "color:pink;" },
  "hi ",
  "mini-react"
);

render(App, document.querySelector("#root"));
```

### v2 拆分框架代码和业务代码

```javascript
// React.js
function createTextNode(text) {
  return {
    type: "TEXT_ELEMENT",
    props: {
      nodeValue: text,
      children: [],
    },
  };
}

function createElement(type, props, ...children) {
  return {
    type,
    props: {
      ...props,
      children: children.map((child) => {
        return typeof child === "string" ? createTextNode(child) : child;
      }),
    },
  };
}

function render(el, container) {
  console.log("render el =>", el);
  console.log("render container =>", container);
  // 1. 判断类型来创建dom
  const dom =
    el.type != "TEXT_ELEMENT"
      ? document.createElement(el.type)
      : document.createTextNode("");

  //  2. 添加props
  Object.keys(el.props).forEach((key) => {
    if (key != "children") {
      dom[key] = el.props[key];
    }
  });
  // 处理children
  const childrens = el.props.children;
  childrens.forEach((child) => {
    render(child, dom);
  });

  // 3. 插入父节点中
  container.append(dom);
}

const React = {
  render,
  createElement,
};

export default React;

// ReactDOM.js
import React from "./React.js";
const ReactDOM = {
  createRoot(container) {
    return {
      render(App) {
        React.render(App, container);
      },
    };
  },
};

export default ReactDOM;

// App.js
import React from "./hreact/core/React.js";

const App = React.createElement(
  "app",
  { id: "app", style: "color:pink;" },
  "hi ",
  "mini-react"
);

export default App;

// main.js

import ReactDOM from "./hreact/core/ReactDOM.js";
import App from "./App.js";
console.log("mini react main start ~");

ReactDOM.createRoot(document.querySelector("#root")).render(App);
```
